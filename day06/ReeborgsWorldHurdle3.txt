def turn_right():
    turn_left()
    turn_left()
    turn_left()


def jump():
    turn_left()
    move()
    turn_right()
    move()
    turn_right()
    move()
    # no need for this turn once the finish is reached
    if not at_goal():
        turn_left()


while not at_goal():
    while front_is_clear():
        move()
    jump()
