# Also valid for Hurdle 1, Hurdle 2 & Hurdle 3.

def turn_right():
    turn_left()
    turn_left()
    turn_left()


def variable_jump():
    turn_left()
    while wall_on_right():
        move()
    turn_right()
    move()
    turn_right()
    # "fall" until you hit the ground
    while front_is_clear():
        move()
    # no need for this turn once the finish is reached
    if not at_goal():
        turn_left()


while not at_goal():
    while front_is_clear():
        move()
    variable_jump()
